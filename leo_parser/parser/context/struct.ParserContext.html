pub struct ParserContext {
    tokens: Vec<SpannedToken>,
    end_span: Span,
    pub(crate) fuzzy_struct_state: bool,
}
Expand description

Stores a program in tokenized format plus additional context. May be converted into a [Program] AST by parsing all tokens.
